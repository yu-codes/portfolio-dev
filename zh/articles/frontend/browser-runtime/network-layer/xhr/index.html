XMLHttpRequest

XMLHttpRequest（XHR）是瀏覽器最早提供的非同步 HTTP 請求機制，是 AJAX（Asynchronous JavaScript and XML）技術的核心。儘管 Fetch API 已成為現代首選，XHR 仍在許多場景中不可或缺——特別是上傳進度監控、舊瀏覽器支援和既有程式碼維護。理解 XHR 的運作原理有助於全面掌握瀏覽器的網路層架構。

核心概念

基本使用流程

const xhr = new XMLHttpRequest();
xhr.open('GET', 'https://api.example.com/data', true); // method, url, async
xhr.setRequestHeader('Content-Type', 'application/json');

xhr.onreadystatechange = function() {
  if (xhr.readyState === 4 && xhr.status === 200) {
    const data = JSON.parse(xhr.responseText);
    console.log(data);
  }
};

xhr.onerror = function() {
  console.error('Network error');
};

xhr.send();

ReadyState 狀態機

XHR 物件的 readyState 屬性反映請求的生命週期階段：

值常數說明
0UNSENTXHR 物件已建立，尚未呼叫 open()
1OPENEDopen() 已呼叫，可設定標頭
2HEADERS_RECEIVEDsend() 已呼叫，已接收回應標頭
3LOADING正在接收回應主體
4DONE操作完成（成功或失敗）

事件系統

XHR 提供豐富的事件監聽機制：

const xhr = new XMLHttpRequest();

// 進度事件
xhr.onloadstart = () => console.log('開始');
xhr.onprogress = (e) => {
  if (e.lengthComputable) {
    console.log(`下載進度: ${(e.loaded / e.total * 100).toFixed(1)}%`);
  }
};
xhr.onload = () => console.log('完成', xhr.status);
xhr.onerror = () => console.log('網路錯誤');
xhr.ontimeout = () => console.log('超時');
xhr.onloadend = () => console.log('結束（無論成功失敗）');
xhr.onabort = () => console.log('已取消');

// 上傳進度事件
xhr.upload.onprogress = (e) => {
  if (e.lengthComputable) {
    console.log(`上傳進度: ${(e.loaded / e.total * 100).toFixed(1)}%`);
  }
};

Response 類型

透過 responseType 屬性指定期望的回應格式：

xhr.responseType = 'json';        // 自動解析 JSON
xhr.responseType = 'blob';        // 二進位 Blob 物件
xhr.responseType = 'arraybuffer'; // ArrayBuffer
xhr.responseType = 'document';    // XML/HTML Document
xhr.responseType = 'text';        // 預設，純文字

超時與取消

const xhr = new XMLHttpRequest();
xhr.open('GET', url);
xhr.timeout = 5000; // 5 秒超時

xhr.ontimeout = () => {
  console.error('請求超時');
};

xhr.send();

// 手動取消
function cancelRequest() {
  xhr.abort();
}

XHR 與事件循環

XHR 的回調函式在事件循環的宏任務佇列（Macrotask Queue）中執行。當網路回應到達時，瀏覽器將對應的回調放入佇列，等待當前執行棧清空後執行。

同步 vs 非同步模式

// 非同步模式（推薦）- 第三個參數為 true 或省略
xhr.open('GET', url, true);

// 同步模式（已棄用）- 會阻塞主執行緒
xhr.open('GET', url, false);
xhr.send();
// 此處可直接使用 xhr.responseText（已同步等待完成）

⚠️ 同步 XHR 在主執行緒上已被棄用（主執行緒外的 Worker 中仍可使用），因為它會凍結整個 UI 直到請求完成。

應用場景

場景一：帶進度的檔案上傳

function uploadWithProgress(file, url, onProgress) {
  return new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest();
    const formData = new FormData();
    formData.append('file', file);

    xhr.upload.addEventListener('progress', (e) => {
      if (e.lengthComputable) {
        const percent = Math.round((e.loaded / e.total) * 100);
        onProgress?.(percent);
      }
    });

    xhr.addEventListener('load', () => {
      if (xhr.status >= 200 && xhr.status < 300) {
        resolve(JSON.parse(xhr.responseText));
      } else {
        reject(new Error(`Upload failed: ${xhr.status}`));
      }
    });

    xhr.addEventListener('error', () => reject(new Error('Network error')));
    xhr.addEventListener('timeout', () => reject(new Error('Timeout')));

    xhr.open('POST', url);
    xhr.timeout = 60000;
    xhr.send(formData);
  });
}

// 使用
uploadWithProgress(file, '/api/upload', (percent) => {
  progressBar.style.width = `${percent}%`;
  progressText.textContent = `${percent}%`;
});

場景二：封裝為 Promise 的通用 XHR

function xhrRequest(options) {
  return new Promise((resolve, reject) => {
    const { method = 'GET', url, headers = {}, body, timeout = 10000, responseType = 'json' } = options;

    const xhr = new XMLHttpRequest();
    xhr.open(method, url);
    xhr.responseType = responseType;
    xhr.timeout = timeout;

    Object.entries(headers).forEach(([key, value]) => {
      xhr.setRequestHeader(key, value);
    });

    xhr.onload = () => {
      if (xhr.status >= 200 && xhr.status < 300) {
        resolve({ status: xhr.status, data: xhr.response, headers: xhr.getAllResponseHeaders() });
      } else {
        reject({ status: xhr.status, data: xhr.response });
      }
    };

    xhr.onerror = () => reject(new Error('Network error'));
    xhr.ontimeout = () => reject(new Error('Request timeout'));

    xhr.send(body ? JSON.stringify(body) : null);
  });
}

場景三：長輪詢（Long Polling）

class LongPolling {
  constructor(url, onMessage) {
    this.url = url;
    this.onMessage = onMessage;
    this.active = true;
    this.poll();
  }

  poll() {
    if (!this.active) return;

    const xhr = new XMLHttpRequest();
    xhr.open('GET', this.url);
    xhr.timeout = 30000;

    xhr.onload = () => {
      if (xhr.status === 200) {
        this.onMessage(JSON.parse(xhr.responseText));
      }
      this.poll(); // 立即發起下一次輪詢
    };

    xhr.ontimeout = () => this.poll(); // 超時後重新連接
    xhr.onerror = () => {
      setTimeout(() => this.poll(), 3000); // 錯誤後延遲重試
    };

    xhr.send();
    this.currentXhr = xhr;
  }

  stop() {
    this.active = false;
    this.currentXhr?.abort();
  }
}

場景四：二進位資料下載

function downloadBinary(url, onProgress) {
  return new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest();
    xhr.open('GET', url);
    xhr.responseType = 'blob';

    xhr.onprogress = (e) => {
      if (e.lengthComputable) {
        onProgress?.({ loaded: e.loaded, total: e.total });
      }
    };

    xhr.onload = () => {
      if (xhr.status === 200) {
        const blob = xhr.response;
        const downloadUrl = URL.createObjectURL(blob);
        resolve({ blob, url: downloadUrl });
      } else {
        reject(new Error(`Download failed: ${xhr.status}`));
      }
    };

    xhr.onerror = () => reject(new Error('Network error'));
    xhr.send();
  });
}
